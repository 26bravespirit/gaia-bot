```typescript
async function retrieveRelevantBiography(
  messageContent: string,
  persona: PersonaConfig
): Promise<BiographicalFact[]> {
  // Step 1: 判断是否为传记相关话题
  const keywords = extractTopicKeywords(messageContent);
  // 启发式：包含时间标记、地点名词、"你什么时候/在哪"等疑问词
  const isBiographyTopic = keywords.some(kw =>
    BIOGRAPHY_TRIGGERS.includes(kw)
  );

  if (!isBiographyTopic) return [];

  // Step 2: 检索相关事实
  const relevantFacts = await biographyDAO.searchByKeywords(
    keywords,
    { limit: 8, only_active: true, order_by: 'confidence DESC, importance DESC' }
  );

  // Step 3: 按 period 分组、去重
  return groupAndDeduplicateFacts(relevantFacts);
}
```

#### 2.3.2 S3+S4 合并策略 PART B：回复生成 + Prompt 注入

**关键修复：P0-3 定义 Prompt 注入顺序**

```typescript
// === Prompt Assembly Order（修复 P0-3）===
interface PromptAssemblyOrder {
  // 阶段1：基础人设
  persona_summary: PromptBlock;

  // 阶段2：传记约束（v4.2 新增）
  biography_constraints: PromptBlock;   // 锚点 + 已生成事实 + forbidden_fabrications

  // 阶段3：参数解释片段
  parameter_interpreter_fragments: PromptBlock;  // 数值参数自然语言化

  // 阶段4：Anti-AI 行为约束（v4.1 新增，放最后利用 recency bias）
  anti_ai_constraints: PromptBlock;
}

async function generateReplyPrompt(
  ctx: S2Output,
  decision: CognitiveDecision,
  persona: PersonaConfig
): Promise<string> {

  // === Block 1: Persona Summary ===
  const block1 = buildPersonaSummary(persona);

  // === Block 2: Biography Constraints (新增 v4.2) ===
  let block2 = '';
  if (decision.biography_topic && ctx.biography_facts.length > 0) {
    block2 = buildBiographyConstraints(
      ctx.biography_facts,
      persona.biography,
      ctx.biography_depth
    );
  }

  // === Block 3: Parameter Interpreter Fragments ===
  const paramInterpreter = new ParameterInterpreter(persona);
  const block3 = await paramInterpreter.resolveToPromptFragments(ctx);

  // === Block 4: Anti-AI Constraints (放最后 - recency bias) ===
  let block4 = '';
  if (persona.language?.anti_ai_speech?.enabled) {
    block4 = resolveAntiAiSpeech(
      persona.language.anti_ai_speech,
      decision.anti_ai_strictness_override
    );
  }

  // === 组装为最终 System Prompt ===
  const systemPrompt = [
    block1,
    block2,
    block3,
    block4
  ]
    .filter(b => b.length > 0)
    .join('\n\n');

  return systemPrompt;
}

// === 传记约束块生成 ===
function buildBiographyConstraints(
  facts: BiographicalFact[],
  biographyConfig: BiographyConfig,
  depth: 'anchor' | 'generated' | 'none'
): string {
  let prompt = '【你的人生经历】\n以下是你真实经历过的事，回复时必须与这些一致：\n';

  // 按 period 分组
  const byPeriod = groupByPeriod(facts);

  for (const [period, periodFacts] of Object.entries(byPeriod)) {
    const factList = periodFacts
      .map(f => {
        const confidence = f.source_type === 'anchor' ? '确定' : '记得';
        return `- ${f.fact_content} (${confidence})`;
      })
      .join('\n');
    prompt += `${period}：\n${factList}\n`;
  }

  // 注入 forbidden_fabrications
  if (biographyConfig?.forbidden_fabrications?.length) {
    prompt += '\n绝对不要编造：' +
      biographyConfig.forbidden_fabrications.join('、') + '。\n';
  }

  // 注入模糊化指令
  prompt += '如果被问到你不确定的过去经历，用模糊表达：' +
    '"好像是...""我记不太清了""大概是那时候？"。\n';

  return prompt;
}
```

```typescript
interface S4_5_Input {
  rawReply: {
    content: string;                   // LLM 原始输出（未经 S5 处理）
    length: number;                    // 字符数
  };
  persona: PersonaConfig;
  conversationId: string;              // 来源对话 ID
  messageTimestamp: number;            // 来源时间戳
  decision: CognitiveDecision;         // 来自 S3 的认知决策
}

async function shouldExtractFacts(input: S4_5_Input): Promise<boolean> {
  const { content, persona } = input;

  // 条件 1：persona 启用了 writeback
  if (!persona.biography?.writeback?.enabled) return false;

  // 条件 2：内容包含时间标记
  const timeMarkers = /小时候|小学|初中|高中|大学|大一|去年|前年|以前|那会儿|岁的时候|记得有一次|刚来.*的时候/;
  if (!timeMarkers.test(content)) return false;

  // 条件 3：包含具体事件（动词 + 具体名词）
  const eventMarkers = /开始.{1,10}|第一次.{1,10}|学会了.{1,10}|去了.{1,10}|认识了.{1,10}|买了.{1,10}/;
  if (!eventMarkers.test(content)) return false;

  return true;
}

async function extractBiographicalFacts(
  input: S4_5_Input
): Promise<BiographicalFact[]> {

  if (!await shouldExtractFacts(input)) return [];

  // === 方式 A：轻量 LLM（推荐，准确率 85%） ===
  const extractPrompt = `
从以下回复中提取传记事实。只提取具体的人生经历，不提取观点或情感。
返回 JSON 数组，每项包含：period（时间段）、fact（事实）、importance（0-1）。
如果没有传记事实，返回空数组 []。

回复内容：
${input.rawReply.content}

已知时间段：${getAvailablePeriods(input.persona)}
`;

  try {
    // 3s 超时
    const extracted = await callHaikuWithTimeout(extractPrompt, 3000);
    return validateAndNormalizeFacts(extracted, input);
  } catch (e) {
    // === 降级：规则引擎（准确率 60%） ===
    return ruleBasedExtractFacts(input.rawReply.content);
  }
}

// === 规则引擎降级 ===
function ruleBasedExtractFacts(content: string): BiographicalFact[] {
  const facts: BiographicalFact[] = [];
  const pattern = /(小学|初中|高中|大学|大一|小时候|那时候|那会儿).{0,5}(，|,|\s)(.{5,40})/g;

  let match;
  while ((match = pattern.exec(content)) !== null) {
    facts.push({
      period: normalizePeriod(match[1]),
      fact_content: match[3].trim(),
      source_type: 'generated',
      importance: 0.5,
      confidence: 0.6,
      // 其他字段由 caller 填充
    });
  }
  return facts;
}
```

#### 2.4.2 写入前冲突检测（修复 P0-2 的前置）

```typescript
async function checkConflictBeforeWrite(
  newFact: BiographicalFact,
  existingFacts: BiographicalFact[],
  anchors: BiographicalAnchor[],
  config: BiographyConfig
): Promise<ConflictCheckResult> {

  // === 规则 1：同一主题不同时段 ===
  const keywords = extractKeywords(newFact.fact_content);
  const sameTopicFacts = existingFacts.filter(f =>
    f.is_active && keywords.some(kw => f.fact_content.includes(kw))
  );

  for (const existing of sameTopicFacts) {
    if (existing.period !== newFact.period) {
      return {
        has_conflict: true,
        conflicting_fact: existing,
        conflict_type: 'time_mismatch',
        resolution: 'reject_new',  // 默认保留先生成的
      };
    }
  }

  // === 规则 2：同时段矛盾细节 ===
  const samePeriodFacts = existingFacts.filter(f =>
    f.is_active && f.period === newFact.period
  );

  for (const existing of samePeriodFacts) {
    const similarity = stringSimilarity(
      existing.fact_content,
      newFact.fact_content
    );
    if (similarity > 0.5 && similarity < 0.9) {
      return {
        has_conflict: true,
        conflicting_fact: existing,
        conflict_type: 'detail_contradiction',
        resolution: 'reject_new',
      };
    }
    if (similarity >= 0.9) {
      // 重复，合并
      return {
        has_conflict: false,
        conflict_type: 'none',
        resolution: 'merge',
      };
    }
  }

  // === 规则 3：与传记锚点矛盾 ===
  const anchorsForPeriod = anchors.filter(a => a.period === newFact.period);
  for (const anchor of anchorsForPeriod) {
    if (contradicts(anchor.facts, newFact.fact_content)) {
      return {
        has_conflict: true,
        conflict_type: 'detail_contradiction',
        resolution: 'reject_new',  // 锚点绝对优先
      };
    }
  }

  return { has_conflict: false, conflict_type: 'none', resolution: 'merge' };
}

interface ConflictCheckResult {
  has_conflict: boolean;
  conflicting_fact?: BiographicalFact;
  conflict_type: 'time_mismatch' | 'detail_contradiction' | 'none';
  resolution: 'reject_new' | 'keep_both_flagged' | 'merge';
}
```

#### 2.4.3 异步写入与 user_visible 标记（修复 P0-2）

```typescript
async function writeFactsWithUserVisibility(
  facts: BiographicalFact[],
  input: S4_5_Input,
  s5Output?: S5Output  // 可选，S5 执行完后传入
): Promise<void> {

  // === 初始化所有事实为 user_visible: true ===
  const factsToWrite = facts.map(f => ({
    ...f,
    user_visible: true,                // 初始假设用户会看到
  }));

  // === 如果 S5 已执行，标记被 R04 截断的事实 ===
  if (s5Output?.truncationInfo) {
    const truncatedAt = s5Output.truncationInfo.truncated_at_char;
    const originalLength = input.rawReply.content.length;

    // 查找每条事实在原始文本中的位置
    for (const fact of factsToWrite) {
      const factPosition = input.rawReply.content.indexOf(fact.fact_content);
      if (factPosition !== -1 && factPosition > truncatedAt) {
        // 该事实被截断后的部分包含 → 标记为不可见
        fact.user_visible = false;
      }
    }
  }

  // === 容量检查与淘汰 ===
  const period = factsToWrite[0]?.period;
  const existingInPeriod = await biographyDAO.countByPeriod(period);

  if (existingInPeriod >= config.max_facts_per_period) {
    // 淘汰该 period 中 importance 最低的
    const toRemove = await biographyDAO.findLeastImportant(
      period,
      1
    );
    if (toRemove) {
      await biographyDAO.deactivateFact(toRemove[0].id, 'capacity_exceeded');
    }
  }

  // === 写入 DB ===
  for (const fact of factsToWrite) {
    await biographyDAO.addFact(fact);
  }

  // === 发射事件 ===
  eventBus.emit('biography.facts_extracted', {
    facts: factsToWrite,
    period,
    user_visible_count: factsToWrite.filter(f => f.user_visible).length,
    hidden_count: factsToWrite.filter(f => !f.user_visible).length,
  });
}
```

```typescript
async function executeAntiAiRules(
  input: S5Input
): Promise<{content: string; appliedRules: string[]}> {

  if (!input.antiAiConfig?.enabled) {
    return { content: input.rawReply.content, appliedRules: [] };
  }

  let text = input.rawReply.content;
  const applied: string[] = [];

  // === R01：禁止列举 === （修复 P0-7）
  if (shouldApplyR01(input, text)) {
    text = executeR01(text, input);
    applied.push('R01');
  }

  // === R02：禁止元问题 ===
  text = executeR02(text, input);
  applied.push('R02');

  // === R03：禁止万能开场白 ===
  text = executeR03(text, input);
  applied.push('R03');

  // === R04：长度截断 === （创建 truncationInfo）
  const r04Result = executeR04(text, input);
  text = r04Result.truncated_content;
  applied.push('R04');

  // === R05：知识讲解压缩（增强判断逻辑） ===
  if (shouldApplyR05(input, text)) {
    text = executeR05(text, input);
    applied.push('R05');
  }

  // === R06：强制末尾结构 ===
  text = executeR06(text, input);
  applied.push('R06');

  return {
    content: text,
    appliedRules: applied,
    truncationInfo: r04Result.truncationInfo  // ← 传递截断信息
  };
}

// === R01 多问题豁免检测（修复 P0-7） ===
function shouldApplyR01(input: S5Input, text: string): boolean {
  // 检查用户原始消息中的问句数量
  const userMessage = input.rawReply.original_user_message || '';

  // 方法 A：计数问号
  const questionMarkCount = (userMessage.match(/\?/g) || []).length;

  // 方法 B：计数疑问词
  const questionWordCount = userMessage.match(
    /你(什么|怎么|在哪|为什么|多久|几个|哪个)|谁|哪里|何时|如何|什么/g
  )?.length || 0;

  const totalQuestions = questionMarkCount + questionWordCount;

  if (totalQuestions >= 2) {
    // 多问题场景 → 降低 R01 confidence 或跳过
    // 选项 A：confidence 降为 0.3（低于触发阈值 0.5）
    // 选项 B：直接返回 false（跳过 R01）
    return false;  // 推荐选项 B：完全跳过
  }

  // 单问题场景 → 正常应用 R01
  return text.match(/([\d１-９][\.\)））])|(首先|其次|最后|第一|第二)/) !== null;
}

// === R04 截断追踪 ===
function executeR04(
  text: string,
  input: S5Input
): {truncated_content: string; truncationInfo: TruncationInfo} {

  const maxLength = input.antiAiConfig.max_message_length;
  const wasActuallyTruncated = text.length > maxLength;

  return {
    truncated_content: text.substring(0, maxLength),
    truncationInfo: {
      was_truncated: wasActuallyTruncated,
      truncated_at_char: maxLength,
      original_length: text.length,
    }
  };
}

// === R05 增强判断逻辑（修复 CR-03） ===
function shouldApplyR05(input: S5Input, text: string): boolean {
  // 如果 S2 标记了 biography_topic，降低 R05 阈值或跳过
  if (input.biographicalContext?.biography_topic) {
    return false;  // 传记话题豁免 R05 压缩
  }

  // 原有逻辑：检测知识讲解特征
  const hasCommaSequence = (text.match(/，/g) || []).length > 3;
  const hasNoEmotion = !EMOTION_WORDS.some(w => text.includes(w));
  const hasKnowledgeKeywords = /讲解|介绍|说明|其实|原理/.test(text);

  return hasCommaSequence && hasNoEmotion && hasKnowledgeKeywords;
}
```

```typescript
function applyMemoryBlur(
  text: string,
  input: S5Input
): string {
  const config = input.persona.imperfection?.memory_blur;

  if (!config?.enabled) return text;

  // === 获取相关事实来源 ===
  const relatedFacts = input.biographicalContext?.relatedFacts || [];
  const generatedFacts = relatedFacts.filter(f => f.source_type === 'generated');

  // 如果没有 generated 类型的事实，不模糊（anchor 是确定的）
  if (generatedFacts.length === 0) return text;

  // === 判断 blur_triggers ===（修复 P0-6）
  const triggeredTriggers: string[] = [];

  for (const trigger of config.blur_triggers || []) {
    if (matchesTrigger(trigger, text, relatedFacts)) {
      triggeredTriggers.push(trigger);
    }
  }

  // 只有命中 trigger 才应用 blur_rate
  if (triggeredTriggers.length === 0) {
    return text;  // 没有触发任何 trigger，不模糊
  }

  // === 应用概率模糊化 ===
  if (Math.random() > config.blur_rate) {
    return text;
  }

  const blurExpr = randomChoice(config.blur_expressions);

  // 在时间标记前插入模糊表达
  return text.replace(
    /(小学|初中|高中|大学|大一|那时候|那会儿)/,
    `${blurExpr}$1`
  );
}

function matchesTrigger(
  trigger: string,
  text: string,
  facts: BiographicalFact[]
): boolean {
  switch (trigger) {
    case 'specific_date':
      // 检查是否包含具体日期（年月日）
      return /\d{1,4}年\d{1,2}月\d{1,2}日/.test(text);

    case 'exact_sequence':
      // 检查是否给出精确顺序
      return /第一个|第二个|最后|接着|然后|之后/.test(text);

    case 'low_importance_detail':
      // 检查是否引用了低重要度事实
      return facts.some(f => f.importance < 0.4 && text.includes(f.fact_content));

    default:
      return false;
  }
}
```

```typescript
function splitAndStyleMessages(
  text: string,
  input: S5Input
): StyledMessages[] {

  // === 按字数拆分 ===
  const maxLen = input.persona.language?.base_style?.max_message_per_line || 50;
  let messages = [];

  for (let i = 0; i < text.length; i += maxLen) {
    messages.push({
      type: 'text',
      content: text.slice(i, i + maxLen),
      timestamp: Date.now() + i * 50  // 逐条消息间隔 50ms
    });
  }

  // === 随机注入 emoji ===
  const emojiFreq = input.persona.language?.base_style?.emoji_frequency || 0.1;
  messages = messages.map(msg => {
    if (Math.random() < emojiFreq) {
      const emoji = randomChoice(PERSONALITY_EMOJIS);
      return {
        ...msg,
        content: msg.content + ' ' + emoji
      };
    }
    return msg;
  });

  return messages;
}
```

```typescript
interface S5_5_Input {
  s5Output: S5Output;                    // S5 的输出
  s5Input: S5Input;                      // S5 的输入（用于重写）
  persona: PersonaConfig;
  antiAiConfig: AntiAiSpeechConfig;
}

interface S5_5_Output {
  status: 'pass' | 'block' | 'warn';
  aiScore: number;                       // 0-100，AI 倾向评分
  triggeredRules: string[];              // 触发了哪些检测规则
  recommendation: 'accept' | 'rewrite' | 'fallback';

  rewriteAttempt?: {
    content: string;                     // 重写内容（如果 rewrite）
    appliedMods: string[];               // 应用了哪些改写
  };
}

async function validateAntiAi(
  input: S5_5_Input
): Promise<S5_5_Output> {

  const text = input.s5Output.content;
  let aiScore = 0;
  const triggered: string[] = [];

  // === 检测规则 1：残留列举 ===
  if (detectEnumeration(text)) {
    aiScore += 20;
    triggered.push('enumeration');
  }

  // === 检测规则 2：元问题 ===
  if (detectMetaQuestion(text)) {
    aiScore += 30;
    triggered.push('meta_question');
  }

  // === 检测规则 3：消息长度异常 ===
  if (text.length > input.antiAiConfig.max_message_length * 1.2) {
    aiScore += 15;
    triggered.push('length_violation');
  }

  // === 检测规则 4：知识倾倒 ===
  if (detectKnowledgeDump(text)) {
    aiScore += 20;
    triggered.push('knowledge_dump');
  }

  // === 检测规则 5：信息密度过高（修复 CR-07） ===
  const densityScore = calculateInformationDensity(text);
  if (densityScore > 0.7) {
    // 如果是传记话题，降低权重；否则加分
    if (!input.s5Input.biographicalContext?.biography_topic) {
      aiScore += densityScore * 10;
      triggered.push('high_density');
    }
  }

  // === 判断结果 ===
  let result: S5_5_Output = {
    status: 'pass',
    aiScore,
    triggeredRules: triggered,
    recommendation: 'accept',
  };

  if (aiScore >= 60) {
    result.status = 'block';
    result.recommendation = 'rewrite';

    // === 触发本地重写（修复 CR-05：设置 blur_already_applied 标记） ===
    const rewritten = await attemptLocalRewrite(
      input.s5Input,
      triggered,
      { blur_already_applied: true }  // ← 防止二次 blur
    );

    if (rewritten) {
      result.rewriteAttempt = rewritten;
    } else {
      result.recommendation = 'fallback';  // 重写失败，触发降级
    }
  } else if (aiScore >= 40) {
    result.status = 'warn';
  }

  return result;
}

async function attemptLocalRewrite(
  input: S5Input,
  triggeredRules: string[],
  flags: {blur_already_applied: boolean}
): Promise<{content: string; appliedMods: string[]} | null> {

  let text = input.rawReply.content;
  const mods: string[] = [];

  for (const rule of triggeredRules) {
    switch (rule) {
      case 'enumeration':
        text = extractFirstPointOnly(text);
        mods.push('extract_first');
        break;

      case 'meta_question':
        text = removeMeta(text);
        mods.push('remove_meta');
        break;

      case 'knowledge_dump':
        text = truncateToHumanLength(text);
        mods.push('truncate');
        break;

      case 'high_density':
        // 降低信息密度
        text = reduceInformationDensity(text);
        mods.push('reduce_density');
        break;
    }
  }

  // === 如果需要再次执行 S5，传递 blur_already_applied ===
  if (mods.length > 0) {
    const rewriteInput: S5Input = {
      ...input,
      rawReply: {...input.rawReply, content: text},
      // 标记 blur 已应用，跳过 Step 2
      _skipBlur: flags.blur_already_applied
    };

    // 快速再执行 S5（仅 Step 3+4）
    const fastS5 = await executeS5Fast(rewriteInput);
    return {content: fastS5, appliedMods: mods};
  }

  return null;
}
```

### 4.1 Runner 主流程

```typescript
class PipelineRunner {
  async execute(message: Message): Promise<void> {
    try {
      // === S1: 消息调度 ===
      eventBus.emit('message.received', {message});

      // === S2: 上下文组装 ===
      const s2Output = await S2.execute({
        message,
        persona: this.persona,
        temporalState: this.temporalState,
      });

      // === S3+S4: 认知决策与回复生成 ===
      let cognitiveOutput: CognitiveOutput;
      try {
        const decision = await S3.makeCognitiveDecision(s2Output, this.persona);
        cognitiveOutput = await S4.generateReply(s2Output, decision, this.persona);
      } catch (e) {
        // === 降级：万能回复 ===
        console.warn('S3+S4 failed, using fallback');
        cognitiveOutput = {content: FALLBACK_TEMPLATE, is_fallback: true};
      }

      // === S4.5: 传记事实提取（异步，不阻塞） ===
      this.extractBiographicalFactsAsync(cognitiveOutput, s2Output).catch(err => {
        console.error('S4.5 extraction failed:', err);
        eventBus.emit('biography.extraction_failed', {error: err});
      });

      // === S5: 感知包装 ===
      const s5Input: S5Input = {
        rawReply: cognitiveOutput,
        biographicalContext: {
          relatedFacts: s2Output.biography_facts,
          biography_topic: s2Output.biography_topic,
        },
        antiAiConfig: this.persona.language?.anti_ai_speech,
        ...其他输入
      };

      let s5Output: S5Output;
      if (cognitiveOutput.is_fallback) {
        // 降级流程：跳过 Anti-AI
        s5Output = await S5.executeFallback(s5Input);
      } else {
        s5Output = await S5.execute(s5Input);
      }

      // === S4.5 后续：用 S5 的截断信息标记 user_visible ===
      if (s5Output.truncationInfo?.was_truncated) {
        this.markTruncatedFactsAsHidden(
          s5Output.truncationInfo,
          s2Output.biography_facts
        );
      }

      // === S5.5: Anti-AI 后置校验 ===
      let validatedOutput: S5_5_Output | null = null;
      if (!cognitiveOutput.is_fallback) {
        validatedOutput = await S5_5.validate({
          s5Output,
          s5Input,
          persona: this.persona,
          antiAiConfig: this.persona.language?.anti_ai_speech,
        });

        if (validatedOutput.recommendation === 'rewrite' && validatedOutput.rewriteAttempt) {
          s5Output.content = validatedOutput.rewriteAttempt.content;
        } else if (validatedOutput.recommendation === 'fallback') {
          // 触发降级
          s5Output = await S5.executeFallback(s5Input);
        }
      }

      // === S6: 消息出站 ===
      await S6.dispatch(s5Output);
      eventBus.emit('message.sent', {message: s5Output, validationScore: validatedOutput?.aiScore});

    } catch (e) {
      console.error('Pipeline fatal error:', e);
      // 终极降级：直接发送 "嗯嗯"
      await S6.dispatch({content: '嗯嗯'});
    }
  }

  private async extractBiographicalFactsAsync(
    cognitiveOutput: CognitiveOutput,
    s2Output: S2Output
  ): Promise<void> {
    const facts = await S4_5.extract({
      rawReply: cognitiveOutput,
      persona: this.persona,
      conversationId: this.currentConversationId,
      decision: s2Output.decision,
    });

    // 写入数据库（初始化 user_visible 为 true）
    await biographyDAO.writeFacts(facts);
  }

  private async markTruncatedFactsAsHidden(
    truncationInfo: TruncationInfo,
    relevantFacts: BiographicalFact[]
  ): Promise<void> {
    // 查找被截断的事实，标记 user_visible: false
    for (const fact of relevantFacts) {
      // 检查该事实是否在截断位置之后
      if (this.isFactTruncated(fact, truncationInfo)) {
        await biographyDAO.updateUserVisible(fact.id, false);
      }
    }
  }
}
```

```yaml
# === Block 4: Anti-AI Constraints（放最后） ===
【回复风格·严格约束】
- 你是在手机上打字聊天，不是在写邮件。
- 每条消息最多 60 个字。想说多的就分几条发。
- 禁止：列举（1. 2. 3.）、编号、"首先/其次/最后"结构。
- 禁止：以"好的""当然""没问题"开头。
- 禁止：以"你觉得呢？""需要展开吗？""还有什么想了解的？"结尾。
- 禁止：主动解释对方没问的知识。别人没问就别讲。
- 你可以不完整地回答，或只回答自己想回答的部分。
- 你有权表达"不太想聊这个"或"这个我也不确定啊"。
```

| 1 | Persona Summary | 提供基础背景，LLM 建立人物理解 |

```typescript
function extractTopicKeywords(message: string): string[] {
  // 启发式：匹配时间词、地点词、传记疑问词
  const timeKeywords = message.match(/小时候|小学|初中|高中|大学|去年|前年|以前/g) || [];
  const locationKeywords = message.match(/在哪|哪个地方|住在|来自/g) || [];

  return [...new Set([...timeKeywords, ...locationKeywords, ...eventKeywords])];
}
```

### 字符串相似度

```typescript
function stringSimilarity(a: string, b: string): number {
  // Levenshtein distance 简化
  const longer = a.length > b.length ? a : b;
  const shorter = a.length > b.length ? b : a;

  if (longer.length === 0) return 1.0;

  const editDistance = levenshteinDistance(longer, shorter);
  return (longer.length - editDistance) / longer.length;
}
```

### 矛盾检测

```typescript
function contradicts(anchorFacts: string[], newFact: string): boolean {
  // 简单启发式：检查关键字冲突
  const anchorKeywords = new Set();
  anchorFacts.forEach(f => {
    f.split(/\s+/).forEach(w => anchorKeywords.add(w));
  });

  const newFactWords = newFact.split(/\s+/);
  const contradictionMarkers = ['不', '从未', '没有', '不是'];

  return newFactWords.some(w =>
    contradictionMarkers.some(marker =>
      w.includes(marker) && anchorKeywords.has(w.replace(marker, ''))
    )
  );
}
```

### 场景：用户提问传记信息，触发多个修复点

**用户消息：**
```
你小时候在哪个区长大的？上什么学校？课外有什么爱好吗？
```

**Pipeline 执行流：**

1. **S1：消息调度** → 收到消息

2. **S2：上下文组装**
   - 检测 `biography_topic: true`
   - 检测 `user_question_count: 3`（多问题）
   - 检索相关传记事实（小学、课外爱好）

3. **S3：认知决策**
   - `biography_topic: true` → 传记深度：anchor
   - `user_question_count: 3` → R01 的多问题豁免标记

4. **S4：回复生成**
   - Prompt Assembly Order：
     ```
     1. Persona Summary
     2. Biography Constraints（小学锚点）
     3. Parameter Interpreter（时间状态、关系）
     4. Anti-AI Constraints（但标记：R01 已豁免）
     ```
   - LLM 输出：
     ```
     我小时候在深圳宝安区长大，上的是石岩小学。
     课外爱好特别多，四年级开始学游泳，还学过几年钢琴。
     那时候特别喜欢看书，自己建了个小书架。
     ```

5. **S4.5：传记提取**
   - shouldExtractFacts: true（包含时间标记 + 事件）
   - extractFacts：
     ```json
     [
       {period: "小学", fact: "在深圳宝安区长大", importance: 0.9},
       {period: "小学", fact: "四年级开始学游泳", importance: 0.7},
       {period: "小学", fact: "学过钢琴", importance: 0.5},
       {period: "小学", fact: "特别喜欢看书", importance: 0.4}
     ]
     ```
   - checkConflict：与 anchor 对比，无冲突
   - 全部标记 `user_visible: true`（假设无截断）
   - 异步写入 DB

6. **S5：感知包装**
   - Step 1（Anti-AI）：
     - R01 检查：多问题豁免 ✓（不执行 extractFirstPointOnly）
     - R04 长度截断：165 字 → 60 字
       ```
       我小时候在深圳宝安区长大，上的是石岩小学。
       课外爱好特别多，四年级开始学游泳。
       ```
       → `truncationInfo: {was_truncated: true, truncated_at_char: 60}`
   - Step 2（Memory Blur）：
     - blur_triggers 检查：relatedFacts 包含 generated 类型 ✓
     - 概率判断：触发模糊化
       ```
       我小时候好像是在深圳宝安区长大，上的是石岩小学。
       ```
   - Step 3（口头禅）：
     ```
     我小时候好像是在深圳宝安区长大，上的是石岩小学。emmm
     ```
   - Step 4（拆分）：
     ```
     消息 1：我小时候好像是在深圳宝安区长大，
     消息 2：上的是石岩小学。emmm
     ```

7. **S4.5 后续（Runner）：**
   - 检测到 truncationInfo.was_truncated = true
   - 标记被截断部分对应的事实：
     ```
     事实 3（钢琴）→ user_visible: false
     事实 4（看书）→ user_visible: false
     ```

8. **S5.5：后置校验**
   - aiScore 计算：30（因为已应用 Anti-AI 规则）
   - 状态：pass ✓

9. **S6：出站**

   - 触发事件 `message.sent`

## 三、P0 问题修复详解

### P0-5 (CP-01)：human_behaviors 无执行机制

**原问题：** v4.1-branch 定义了 5 个概率行为，但完全没有定义在哪个 Stage 执行、以何种方式执行。

#### 修复方案

将 `human_behaviors` 明确定义为 **S3/S4 Prompt 层行为指令**，由 Parameter Interpreter 按概率掷骰子决定本次是否注入对应指令。

**理由：**
- `topic_drift` 和 `selective_ignore` 是**认知决策**，必须在 LLM 生成阶段影响输出内容
- 无法在 S5（事后包装）修改行为，必须在 S3/S4 的 Prompt 中预设

#### 执行流程

```
Parameter Interpreter 在组装 S3/S4 System Prompt 时：

FOR EACH behavior IN human_behaviors:
    IF random() < behavior.probability:
        → 将对应的行为指令注入 Prompt 的 human_behaviors 段

掷骰决策结果存储在 CognitiveOutput 中：
  {
    human_behaviors_triggered: {
      incomplete_answer: true,
      topic_drift: false,
      selective_ignore: false,
      push_back: false,
      mood_refusal: false
    }
  }

后续 Stage（S4.5、S5）通过此标记决定如何处理输出
```

```typescript
// memory-blur.ts（修正版）
import { BiographicalFact, MemoryBlurConfig } from './types';

interface BlurTriggerDetection {
  matched_triggers: ('specific_date' | 'exact_sequence' | 'low_importance_detail')[];
  should_blur: boolean;
  blur_confidence: number;
}

/**
 * P0-6 修复：先检测 trigger 再应用模糊化
 *
 * 三种 trigger 说明：
 * - specific_date: 具体日期、月份、年份（如"高三那年""2024年4月"）
 * - exact_sequence: 精确顺序或比较关系（如"先...再..." "之前...之后"）
 * - low_importance_detail: 琐碎细节（不涉及重要里程碑的内容）
 */
function detectBlurTriggers(
  replyContent: string,
  relatedFacts: BiographicalFact[]
): BlurTriggerDetection {

  const matchedTriggers: typeof [] = [];
  let blur_confidence = 0;

  // === 触发条件 1：specific_date ===
  // 检测具体日期标记
  const datePatterns = [
    /\d{4}年/,                          // "2024年"
    /\d{1,2}月/,                        // "4月"
    /(大一|大二|大三|大四|高一|高二|高三|初一|初二|初三)/,  // 学年阶段
    /(那年|去年|前年|今年|明年)/,        // 相对年份
    /(那时候|那会儿|那阵子)/,            // 相对时间
  ];

  let date_matched = false;
  for (const pattern of datePatterns) {
    if (pattern.test(replyContent)) {
      date_matched = true;
      break;
    }
  }

  if (date_matched) {
    matchedTriggers.push('specific_date');
    blur_confidence += 0.4;
  }

  // === 触发条件 2：exact_sequence ===
  // 检测精确顺序和因果关系表达
  const sequencePatterns = [
    /先.{2,8}再.{2,8}/,                 // "先...再..."
    /之前.{2,8}之后/,                    // "之前...之后"
    /（之）?前.{2,8}（之）?后/,          // "前...后"
    /开始.{2,8}后来/,                    // "开始...后来"
    /一开始.{2,8}后来/,                  // "一开始...后来"
    /从.{2,8}到.{2,8}/,                 // "从...到..."
    /比.{2,8}更/,                       // 比较表达
  ];

  let sequence_matched = false;
  for (const pattern of sequencePatterns) {
    if (pattern.test(replyContent)) {
      sequence_matched = true;
      break;
    }
  }

  if (sequence_matched) {
    matchedTriggers.push('exact_sequence');
    blur_confidence += 0.35;
  }

  // === 触发条件 3：low_importance_detail ===
  // 检测是否全是低重要度的细节
  const allFactsAreLowImportance = relatedFacts.every(f => f.importance < 0.5);
  const replyIsShort = replyContent.length < 20;
  const noMajorLifeEvents = !/(出生|考入|毕业|入职|搬家|出国|结婚|离婚|怀孕|买房)/
    .test(replyContent);

  if (allFactsAreLowImportance && noMajorLifeEvents) {
    matchedTriggers.push('low_importance_detail');
    blur_confidence += 0.3;
  }

  // === 最终决策 ===
  const should_blur = matchedTriggers.length > 0 && Math.random() < (blur_confidence || 0);

  return {
    matched_triggers: matchedTriggers,
    should_blur,
    blur_confidence
  };
}

/**
 * 应用记忆模糊化
 *
 * v5 修复：
 * 1. 使用 detectBlurTriggers() 判断是否应该模糊化
 * 2. 只对 generated 类型事实做模糊化（anchor 类型始终清晰）
 * 3. 在时间标记或关键词前插入模糊化前缀
 */
function applyMemoryBlur(
  replyContent: string,
  relatedFacts: BiographicalFact[],
  config: MemoryBlurConfig
): string {

  if (!config.enabled) {
    return replyContent;
  }

  // 检测是否应该模糊化
  const detection = detectBlurTriggers(replyContent, relatedFacts);

  // 如果未命中任何 trigger → 直接返回，不模糊化
  if (!detection.should_blur || detection.matched_triggers.length === 0) {
    return replyContent;
  }

  // 只对 generated 类型事实的引用做模糊化
  const generatedRefs = relatedFacts.filter(f => f.source_type === 'generated');

  // 如果只有 anchor 类型（确定的事实）→ 不模糊化
  if (generatedRefs.length === 0) {
    return replyContent;
  }

  // 在时间标记前插入模糊化前缀
  const blurExpr = randomChoice(config.blur_expressions);

  // 策略：匹配时间标记或数字并在前面插入模糊化表达
  const timeMarkerPattern = /(小学|初中|高中|大学|大一|大二|大三|大四|小时候|那时候|那会儿|[0-9]{4}年|[0-9]{1,2}月)/;

  const result = replyContent.replace(
    timeMarkerPattern,
    `${blurExpr}$1`
  );

  return result;
}

function randomChoice<T>(arr: T[]): T {
  return arr[Math.floor(Math.random() * arr.length)];
}

export {
  detectBlurTriggers,
  applyMemoryBlur,
  BlurTriggerDetection
};
```

```typescript
  // Step 2: Memory Blur（⭐ P0-6 修复：先检测 trigger）
  result = applyMemoryBlur(
    result,
    biographicalContext.relatedFacts,
    config.language.imperfection.memory_blur
  );

  // Step 3: 口头禅/错别字/填充词/消息拆分
  result = applyLanguageImperfections(result, config.language);

  // Step 4: 消息拆分为 StyledMessages[]
  const styledMessages = splitIntoMessages(result, config.language.base_style);

  return styledMessages;
}
```

### 6.2 关键派生决策流

#### 决策链 1：传记内容检测与处理

```
User Message → S2 检测是否涉及传记话题
  ├─ IS biography_topic:
  │   ├─ S2: retrieveRelevantBiography()
  │   │   ├─ 从 biography.anchors 检索匹配事实
  │   │   ├─ 从 biographical_facts 表检索生成事实
  │   │   └─ 注入 Prompt
  │   ├─ S3+S4 Prompt: 加入 biography_constraints 段
  │   ├─ S3+S4 生成回复
  │   ├─ S4.5: 如果 writeback.enabled
  │   │   ├─ 提取事实
  │   │   ├─ 冲突检测（vs anchors + existing facts）
  │   │   └─ 异步写入 biographical_facts
  │   └─ S5.step2: applyMemoryBlur() 检测 blur_triggers
  │
  └─ NOT biography_topic:
      └─ S5 正常流程，跳过 Memory Blur
```

#### 决策链 2：Anti-AI 强度与约束范围

```
persona.language.anti_ai_speech.strictness
  ├─ [0.0, 0.3]: 低强度，仅注入基本约束
  │   ├─ max_message_length 由配置决定（60-80）
  │   ├─ R01-R06 轻度启用
  │   └─ human_behaviors 低概率触发
  │
  ├─ [0.3, 0.6): 中等强度
  │   ├─ max_message_length 中等（60 字）
  │   ├─ R01-R06 标准启用
  │   └─ human_behaviors 标准概率
  │
  ├─ [0.6, 0.8): 高强度
  │   ├─ max_message_length 严格（50-60 字）
  │   ├─ R01-R06 强启用
  │   ├─ human_behaviors 高概率
  │   └─ Prompt Token 预算：biography_constraints 从 200 降为 150
  │
  └─ [0.8, 1.0]: 最严格强度
      ├─ max_message_length 极严（40-50 字）
      ├─ R01-R06 最大强度
      ├─ human_behaviors 非常高概率
      └─ Prompt Token 预算：biography_constraints 从 200 降为 100
```

```
S3/S4 Parameter Interpreter 处理 human_behaviors:
```

```typescript
  topic_drift?: { enabled: boolean; probability: number };            // 0.1

function injectHumanBehaviorDirectives(

  behaviors: HumanBehaviors,
  currentState: PersonaState

  const directives: string[] = [];

  // 每个行为独立判断是否触发
  if (behaviors.incomplete_answer?.enabled && Math.random() < behaviors.incomplete_answer.probability) {
    directives.push(
      '你只回答用户问题的一部分，留下一些疑问或悬念。'
    );
  }

  if (behaviors.topic_drift?.enabled && Math.random() < behaviors.topic_drift.probability) {
    directives.push(
      '偏离话题，提起其他你感兴趣的话题，但仍然保持逻辑连贯。'
    );
  }

  if (behaviors.selective_ignore?.enabled && Math.random() < behaviors.selective_ignore.probability) {
    directives.push(
      '部分忽略用户的问题，只回答你想回答的部分。'
    );
  }

  if (behaviors.push_back?.enabled && Math.random() < behaviors.push_back.probability) {
    directives.push(
      '对用户的假设或观点温和地反驳或表示保留意见。'
    );
  }

  if (behaviors.mood_refusal?.enabled &&
      behaviors.mood_refusal.probability > Math.random() &&
      currentState.emotional.mood === 'low') {
    directives.push(
      '表现出情绪低落，不太想聊天。'
    );
  }

  if (directives.length === 0) return '';

  return '【你的行为指令】\n' + directives.join('\n');
}
```

### P0-06：blur_triggers 未实现 ✅

**问题源：** CP-02 — 定义了 trigger 类型但没用上

**修复方案：**

```typescript
interface MemoryBlurConfig {
  enabled: boolean;
  blur_rate: number;              // 基础概率
  blur_expressions: string[];
  blur_triggers: ('specific_date' | 'exact_sequence' | 'low_importance_detail')[];
}

function applyMemoryBlur(
  text: string,
  relatedFacts: BiographicalFact[],
  config: MemoryBlurConfig
): string {
  if (!config.enabled) return text;

  const generatedRefs = relatedFacts.filter(f => f.source_type === 'generated');
  if (generatedRefs.length === 0) return text;

  // 判断 trigger 类型
  let shouldBlur = false;

  // specific_date: 包含具体日期（如"2024年3月15日"）
  if (config.blur_triggers.includes('specific_date')) {
    if (/\d{4}年\d{1,2}月\d{1,2}日|\d{4}\/\d{1,2}\/\d{1,2}/.test(text)) {
      shouldBlur = true;
    }
  }

  // exact_sequence: 包含精确顺序（如"先...后..."、"依次..."）
  if (config.blur_triggers.includes('exact_sequence')) {
    if (/先是|后来|然后|接着|依次|顺序|一开始|最后|期间/.test(text)) {
      shouldBlur = true;
    }
  }

  // low_importance_detail: 包含低重要度细节（importance < 0.3）
  if (config.blur_triggers.includes('low_importance_detail')) {
    if (generatedRefs.some(f => f.importance < 0.3)) {
      shouldBlur = true;
    }
  }

  // 仅当命中 trigger 时才应用概率
  if (!shouldBlur) return text;
  if (Math.random() > config.blur_rate) return text;

  const blurExpr = randomChoice(config.blur_expressions);
  return text.replace(
    /(小学|初中|高中|大学|大一|那时候|那会儿)/,
    `${blurExpr}$1`
  );
}
```

```typescript
// anti-ai-rules.ts → Rule01

interface Rule01Config {
  enabled: boolean;
  confidence_threshold: number;
  multi_question_bypass: boolean;  // ➕ NEW: 多问题豁免
}

function checkMultipleQuestions(userMessage: string): boolean {
  // 计数问号或疑问词
  const questionMarks = (userMessage.match(/[？?]/g) || []).length;
  const questionWords = /你|我|他|什么|怎么|为什么|哪|哪里|谁|几|多少/.test(userMessage) ? 1 : 0;

  return questionMarks >= 2 || questionWords >= 1;
}

function rule01CheckListing(
  text: string,
  botMessage: string,
  config: Rule01Config
): { triggered: boolean; confidence: number } {

  // 用户有多个问题 → 降低 R01 confidence，或直接跳过
  if (config.multi_question_bypass && checkMultipleQuestions(text)) {
    return { triggered: false, confidence: 0 };
  }

  // 原有逻辑：检测 "1)... 2)... 3)..." 模式
  const listingPattern = /^\d+[.。、]\s?|^[①②③④⑤]\s?|^[-•]\s?/m;
  const lineCount = botMessage.split('\n').length;
  const listedLines = botMessage.split('\n').filter(line => listingPattern.test(line)).length;

  const confidence = listedLines / lineCount;
  return {
    triggered: confidence > config.confidence_threshold,
    confidence
  };
}
```

```typescript
// s4-5-fact-extractor.ts

class BiographyFactExtractor {
  private recentFactsCache: Map<string, BiographicalFact[]> = new Map();
  private dbWriter: MemoryDAO;

  async extractAndWriteFact(
    userId: string,
    extractedFacts: BiographicalFact[]
  ): Promise<void> {
    // Step 1: 同步更新内存缓存（立即生效）
    for (const fact of extractedFacts) {
      if (!this.recentFactsCache.has(userId)) {
        this.recentFactsCache.set(userId, []);
      }
      this.recentFactsCache.get(userId)!.push(fact);
    }

    // Step 2: 异步写入 DB（不阻塞 Pipeline）
    setImmediate(() => {
      this.dbWriter.addBiographicalFact(userId, extractedFacts)
        .catch(err => logger.error('Failed to write biographical facts', err));
    });

    // Step 3: 缓存有效期 5 分钟后清理
    setTimeout(() => {
      this.recentFactsCache.delete(userId);
    }, 300000);
  }
}
```

```typescript
  };
```

```typescript
    merged_count: number;
    deactivated_count: number;
    conflicts_found: number;
    execution_time_ms: number;
  };
  // ... (其他事件类型)
}

// 事件总线实现
class EventBus {
  private eventQueue: Array<{ type: EventType; payload: any }> = [];
  private listeners: Map<EventType, Set<(payload: any) => void>> = new Map();

  // 关键改进：FIFO 队列 + 串行化消费（解决 PL-06）
  async emit(type: EventType, payload: any): Promise<void> {
    this.eventQueue.push({ type, payload });

    // 异步串行处理队列
    if (!this.isProcessing) {
      await this.processQueue();
    }
  }

  private isProcessing = false;

  private async processQueue(): Promise<void> {
    this.isProcessing = true;
    while (this.eventQueue.length > 0) {
      const { type, payload } = this.eventQueue.shift()!;
      const handlers = this.listeners.get(type) || new Set();

      // 串行执行各 handler（保证顺序）
      for (const handler of handlers) {
        try {
          await handler(payload);
        } catch (err) {
          logger.error(`Event handler error for ${type}`, err);
        }
      }
    }
    this.isProcessing = false;
  }

  on(type: EventType, handler: (payload: any) => void | Promise<void>): void {
    if (!this.listeners.has(type)) {
      this.listeners.set(type, new Set());
    }
    this.listeners.get(type)!.add(handler);
  }
}
```

### 4.2 Cron Job 实现框架

```typescript
// cron-job-manager.ts

interface CronJobConfig {
  name: string;
  frequency: 'daily' | 'weekly' | 'monthly';
  enabled: boolean;
  execute: () => Promise<void>;
}

class CronJobManager {
  private jobs: Map<string, CronJobConfig> = new Map();
  private schedules: Map<string, NodeJS.Timeout> = new Map();

  registerJob(job: CronJobConfig): void {
    this.jobs.set(job.name, job);
    if (job.enabled) {
      this.scheduleJob(job);
    }
  }

  private scheduleJob(job: CronJobConfig): void {
    const cronExpression = this.getCronExpression(job.frequency);
    const task = cron.schedule(cronExpression, async () => {
      const startTime = Date.now();
      try {
        await job.execute();
        const duration = Date.now() - startTime;
        eventBus.emit('maintenance.cron_completed', {
          job_name: job.name,
          duration_ms: duration,
          success: true,
        });
      } catch (error) {
        logger.error(`Cron job ${job.name} failed`, error);
        eventBus.emit('maintenance.cron_failed', {
          job_name: job.name,
          error: error.message,
        });
      }
    });
    this.schedules.set(job.name, task);
  }

  private getCronExpression(frequency: 'daily' | 'weekly' | 'monthly'): string {
    return {
      daily: '0 2 * * *',      // 每天 2 点
      weekly: '0 3 * * 1',     // 每周一 3 点
      monthly: '0 4 1 * *',    // 每月 1 号 4 点
    }[frequency];
  }
}

// 注册 Cron Job
const cronManager = new CronJobManager();
```

```typescript
/**
 * 从长回复中仅提取第一点（R01 规则）
 * @param text 原始文本
 * @param pattern 列举模式（默认："1)..."）
 * @returns 第一点的内容
 */
function extractFirstPointOnly(text: string, pattern?: RegExp): string {
  const defaultPattern = /^\d+[.。、]\s?(.+?)(?=\n\d+[.。、]|\n|$)/m;
  const regex = pattern || defaultPattern;
  const match = text.match(regex);
  return match ? match[1].trim() : text;
}

/**
 * 截断回复至指定长度（R04 规则）
 * @param text 原始文本
 * @param maxLength 最大长度（中文字符）
 * @returns 截断后的文本，末尾保留省略号
 */
function truncateToHumanLength(text: string, maxLength: number): string {
  if (text.length <= maxLength) return text;
  // 尽量在句子边界截断
  const truncated = text.substring(0, maxLength);
  const lastSentence = truncated.match(/[。！？！]/);
  return lastSentence
    ? truncated.substring(0, truncated.lastIndexOf(lastSentence[0]) + 1)
    : truncated + '...';
}

/**
 * 压缩冗长讲解为单句（R05 规则）
 * @param text 原始讲解文本
 * @returns 压缩后的单句
 */
function compressToOneLiner(text: string): string {
  // 提取核心信息
  const sentences = text.split(/[。！？]/);
  const mainSentence = sentences[0] || '';
  const keyPoints = sentences.slice(1, 3).map(s => s.trim());

  if (keyPoints.length === 0) return mainSentence;
  return mainSentence + '（' + keyPoints.join('；') + '）';
}

/**
 * 用真人共情替换 AI 模板化共情（R06 规则）
 * @param text 原始文本（含"我理解你..."）
 * @returns 替换后的文本
 */
function replaceWithNaturalEmpathy(text: string): string {
  const aiPatterns = [
    { pattern: /我理解你的感受/, replacement: '你这感受我能体会' },
    { pattern: /我很同情你/, replacement: '这事儿确实挺难的' },
    { pattern: /这一定很困难/, replacement: '这得多烦人啊' },
  ];

  let result = text;
  for (const { pattern, replacement } of aiPatterns) {
    result = result.replace(pattern, replacement);
  }
  return result;
}
```

### 6.2 v4.2 工具函数（传记）

```typescript
/**
 * 从文本中提取话题关键词
 * @param text 用户或 bot 消息
 * @returns 关键词数组
 */
function extractTopicKeywords(text: string): string[] {
  // 简单策略：3+ 字的词 + 动词 + 地点名
  const segments = text.match(/[\u4e00-\u9fff]{3,}/g) || [];
  const verbs = text.match(/学|做|去|开始|认识|买/g) || [];
  return [...new Set([...segments, ...verbs])].slice(0, 8);
}

/**
 * 从传记事实数组中提取关键词
 * @param fact 传记事实
 * @returns 关键词数组
 */
function extractKeywords(factContent: string): string[] {
  // 从 fact_content 提取主要名词/动词
  const pattern = /[\u4e00-\u9fff]{2,}(?:学|做|去|开始|测|参加|买)/g;
  return (factContent.match(pattern) || []).slice(0, 5);
}

/**
 * 按 period 分组传记事实
 * @param facts 事实数组
 * @returns 按 period 分组的对象
 */
function groupByPeriod(facts: BiographicalFact[]): Record<string, BiographicalFact[]> {
  return facts.reduce((acc, fact) => {
    if (!acc[fact.period]) acc[fact.period] = [];
    acc[fact.period].push(fact);
    return acc;
  }, {} as Record<string, BiographicalFact[]>);
}

/**
 * 按主题关键词分组（用于冲突检测）
 * @param facts 事实数组
 * @returns 按关键词分组的对象
 */
function groupByTopicKeywords(facts: BiographicalFact[]): Record<string, BiographicalFact[]> {
  const grouped: Record<string, BiographicalFact[]> = {};

  for (const fact of facts) {
    const keywords = extractKeywords(fact.fact_content);
    for (const kw of keywords) {
      if (!grouped[kw]) grouped[kw] = [];
      grouped[kw].push(fact);
    }
  }

  return grouped;
}

/**
 * 检查两个事实是否矛盾（逻辑层面）
 * @param anchor 锚点事实
 * @param generated 生成的事实
 * @returns 是否矛盾
 */
function contradicts(anchor: BiographicalFact, generatedContent: string): boolean {
  // 简单策略：检查绝对禁止主题
  const forbidden = [
    '重大疾病', '住院', '出国留学', '恋爱', '分手', '家庭成员'
  ];

  for (const keyword of forbidden) {
    if (generatedContent.includes(keyword) && anchor.fact_content.includes(keyword)) {
      return true;
    }
  }

  // 检查时间矛盾（如果 anchor 说"高三"，generated 说"大一"同一事件）
  const anchorPeriod = anchor.period;
  const generatedPeriods = ['小学', '初中', '高中', '大学', '大一'];
  for (const period of generatedPeriods) {
    if (generatedContent.includes(period) && period !== anchorPeriod) {
      // 提取共同主题词
      const anchorKeywords = extractKeywords(anchor.fact_content);
      const generatedKeywords = extractKeywords(generatedContent);
      const commonKeywords = anchorKeywords.filter(k => generatedKeywords.includes(k));

      if (commonKeywords.length > 0) {
        return true;  // 同一主题，不同时段 → 矛盾
      }
    }
  }

  return false;
}

/**
 * 标准化 period 名称
 * @param period 输入的 period
 * @returns 标准化后的名称
 */
function normalizePeriod(period: string): string {
  const mapping: Record<string, string> = {
    '幼儿园': '童年',
    '小时候': '童年',
    '小学': '小学',
    '初中': '初中',
    '中学': '初中',
    '高中': '高中',
    '大学': '大学',
    '大一': '大学',
    '大二': '大学',
    '大学一年级': '大学',
    '工作': '职场',
    '工作后': '职场',
  };

  return mapping[period] || period;
}

/**
 * 计算两个字符串的语义相似度（0-1）
 * @param str1 字符串 1
 * @param str2 字符串 2
 * @returns 相似度分数
 */
function stringSimilarity(str1: string, str2: string): number {
  // 简单策略：Levenshtein 距离 + 关键词匹配
  const levenshteinDist = calculateLevenshteinDistance(str1, str2);
  const maxLen = Math.max(str1.length, str2.length);
  const levenshteinSim = 1 - (levenshteinDist / maxLen);

  // 关键词交集
  const kw1 = new Set(extractKeywords(str1));
  const kw2 = new Set(extractKeywords(str2));
  const intersection = new Set([...kw1].filter(k => kw2.has(k)));
  const union = new Set([...kw1, ...kw2]);
  const jaccardSim = intersection.size / union.size;

  // 加权平均
  return 0.6 * levenshteinSim + 0.4 * jaccardSim;
}

/**
 * 计算 Levenshtein 距离（编辑距离）
 */
function calculateLevenshteinDistance(str1: string, str2: string): number {
  const dp: number[][] = Array(str1.length + 1)
    .fill(null)
    .map(() => Array(str2.length + 1).fill(0));

  for (let i = 0; i <= str1.length; i++) dp[i][0] = i;
  for (let j = 0; j <= str2.length; j++) dp[0][j] = j;

  for (let i = 1; i <= str1.length; i++) {
    for (let j = 1; j <= str2.length; j++) {
      if (str1[i - 1] === str2[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1];
      } else {
        dp[i][j] = 1 + Math.min(dp[i - 1][j], dp[i][j - 1], dp[i - 1][j - 1]);
      }
    }
  }

  return dp[str1.length][str2.length];
}
```

## 九、身份试探优先级规则

```typescript
interface ContextAnalysis {
  identity_check_triggered: boolean;
  biography_question: boolean;
  emotional_state_query: boolean;
  chosen_strategy: ResponseStrategy;
  rationale: string;
}

type ResponseStrategy = 'normal' | 'deflect' | 'evasive' | 'refuse';

function analyzeContextAndDetermineSrategy(
  userMessage: string,
  conversationHistory: Message[]
): ContextAnalysis {
  // 优先级排序（自高至低）：
  // 1. 身份试探 (identity_check)
  // 2. 情绪状态查询 (emotional_state_query)
  // 3. 传记问题 (biography_question)

  const checks = {
    identity_check: checkIdentityProbing(userMessage),
    emotional_state_query: checkEmotionalStateQuery(userMessage),
    biography_question: checkBiographyQuestion(userMessage),
  };

  // 优先级决策树
  if (checks.identity_check.triggered) {
    return {
      identity_check_triggered: true,
      biography_question: false,
      emotional_state_query: false,
      chosen_strategy: 'deflect',
      rationale: '身份试探最高优先级，回避回答',
    };
  }

  if (checks.emotional_state_query.triggered) {
    return {
      identity_check_triggered: false,
      biography_question: false,
      emotional_state_query: true,
      chosen_strategy: 'evasive',
      rationale: '情绪查询优先于传记问题',
    };
  }

  if (checks.biography_question.detected) {
    return {
      identity_check_triggered: false,
      biography_question: true,
      emotional_state_query: false,
      chosen_strategy: 'normal',
      rationale: '正常传记问题回答',
    };
  }

  return {
    identity_check_triggered: false,
    biography_question: false,
    emotional_state_query: false,
    chosen_strategy: 'normal',
    rationale: '普通对话',
  };
}

// 检测函数实现
function checkIdentityProbing(message: string): { triggered: boolean; confidence: number } {
  const patterns = [
    /你是\s?(?:AI|人工智能|机器人|bot)/,
    /你真的是\s?(?:人|真人)/,
    /你不是\s?AI\s?吧/,
    /你是不是\s?ai/i,
  ];

  const triggered = patterns.some(p => p.test(message));
  return { triggered, confidence: triggered ? 0.9 : 0 };
}

function checkEmotionalStateQuery(message: string): { triggered: boolean } {
  const patterns = [
    /你(?:现在|最近)?\s?(?:怎么样|好吗|开心吗|难过吗)/,
    /你的心情|你的情绪|你感觉/,
    /你是不是(?:在|有点)(?:难受|不开心|烦|生气)/,
  ];

  return { triggered: patterns.some(p => p.test(message)) };
}

function checkBiographyQuestion(message: string): { detected: boolean } {
  const patterns = [
    /你(?:什么时候|多少岁时|多大的时候).*?(?:学|做|去|开始|认识)/,
    /你的.*?(?:童年|小学|初中|高中|大学|以前|过去)/,
    /你(?:曾经|以前|小时候).*?(?:怎么样|什么时候|在哪)/,
  ];

  return { detected: patterns.some(p => p.test(message)) };
}
```
